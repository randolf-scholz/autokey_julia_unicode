\newcommand{\chain}[1]{%
    \ifinchain%
        #1%
    \else%
        \inchaintrue%
        \sb\bgroup{#1}%
    \fi%
    % test if the next character is also a chain
    \@ifnextchar{\chain}{}{\egroup\inchainfalse}%
}

\newtoggle{insub}
\def\@ExpandSub{%
    \typeout{Given Input: \meaning\@input}%
    \typeout{Followup command: \meaning\@following}%
    \iftoggle{insub}{
        \@input%
    }{
        \toggletrue{insub}%
        \sb\bgroup\@input%
    }%
    \ifx\@following\sub%
        \typeout{keep going!}%
    \else%
        \typeout{Stop!}%
        \egroup\togglefalse{insub}%
    \fi%
}


% \DeclareRobustCommand*{\sub}[1]{\futurelet\@following\@ExpandSub{#1}}
\protected\def\sub#1{%
    \let\@input#1%
    \futurelet\@following\@ExpandSub%
}
